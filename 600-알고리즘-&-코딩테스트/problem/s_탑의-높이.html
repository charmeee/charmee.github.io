def addHeight(tarr,idx,k):
if k<0:
return -1
tarr[idx] += 1
k -= 1
left = idx - 1
right = idx + 1
if left >= 0 and tarr[idx]-tarr[left]>1:
return addHeight(tarr,left,k)
if right < len(tarr) and tarr[idx]-tarr[right]>1:
return addHeight(tarr,right,k)
return k
